00138 inline void sorted_stack_support::push(size_type x){
00139         x += 1;
00140         ++m_cnt;        //< increment counter
00141         size_type bn = block_nr(x);
00142         m_stack[bn] ^= (1ULL << block_pos(x));
00143         if( bn > 0 and m_stack[bn-1] == 0 ){
00144                 m_stack[bn-1] = 0x8000000000000000ULL | m_top;

00149 inline void sorted_stack_support::pop(){

00151                 --m_cnt; //< decrement counter 
00152                 size_type bn = block_nr(m_top);
00153                 uint64_t w = m_stack[ bn ];     
00154                 assert( (w>>63) == 0 );  // highest bit is not set, as the block contains no pointer
00155                 w ^= ( 1ULL << block_pos(m_top) );
00156                 m_stack[ bn ] = w;

00158                         m_top = bn*63 + bit_magic::l1BP(w);
