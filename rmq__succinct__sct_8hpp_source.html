00129         size_type operator()(const size_type l, const size_type r)const {
00130             assert(l <= r); assert(r < size());
00131             if (l==r)
00132                 return l;
00133             size_type i         = m_sct_bp_support.select(l+1);
00134             size_type j         = m_sct_bp_support.select(r+1);
00135             size_type fc_i      = m_sct_bp_support.find_close(i);
00136             if (j < fc_i) { // i < j < find_close(j) < find_close(i)
00137                 return l;
