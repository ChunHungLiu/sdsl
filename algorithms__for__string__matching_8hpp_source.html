00057         template<class Csa>
00058         static typename Csa::csa_size_type backward_search(
00059                         const Csa &csa, 
00060                         typename Csa::size_type l,
00061                         typename Csa::size_type r,
00062                         typename Csa::char_type c,
00063                         typename Csa::size_type &l_res,
00064                         typename Csa::size_type &r_res
00065                         ){
00066                 assert( l <= r ); assert( r < csa.size() );
00067                 typename Csa::size_type c_begin = csa.C[csa.char2comp[c]]; 
00068                 l_res = c_begin + csa.rank_bwt(l, c); // count c in bwt[0..l-1]
00069                 r_res = c_begin + csa.rank_bwt(r+1, c) - 1; // count c in bwt[0..r]
00070                 assert( r_res+1-l_res >= 0 );
00071                 return r_res+1-l_res;
00072         }

00226         typename Cst::cst_size_type forward_search(const Cst &cst, typename Cst::node_type &v,
00227                                 const typename Cst::size_type d, const typename Cst::char_type c,
00228                                 typename Cst::size_type &char_pos)
00229         {
00230                 unsigned char cc = cst.csa.char2comp[c]; // check if c occures in the text of the csa
00231                 if(cc==0 and cc!=c)                                              //   "    " "    "     "  "    "   "  "   "
00232                         return 0;
00233                 typename Cst::size_type depth_node = cst.depth(v);
00234                 if( d < depth_node  ){ // 
00235                         char_pos = cst.csa.psi[char_pos];
00236                         if( char_pos < cst.csa.C[cc] or char_pos >= cst.csa.C[cc+1] )
00237                                 return 0;
00238                         return cst.leaves_in_the_subtree(v);

00240                         v = cst.child(v, c, char_pos);

00244                                 return cst.leaves_in_the_subtree(v);
