00041 template<class tCst>
00042 double H0(const typename tCst::node_type &v, const tCst &cst){
00043         if( cst.is_leaf(v) ){
00044                 return 0;
00045         }else{
00046                 double h0=0;
00047                 typename tCst::size_type n = cst.leaves_in_the_subtree(v);
00048                 typename tCst::node_type w = cst.ith_child(v, 1);
00049                 do{
00050                         double p = ((double)cst.leaves_in_the_subtree(w))/n;
00051                         h0 -= p*log2(p);
00052                         w = cst.sibling(w);
00053                 }
00054                 while( w != cst.root() );
00055                 return h0;
00056         }
00057 }

00066 double Hk(const tCst &cst, typename tCst::size_type k, typename tCst::size_type &context){
00067         double hk = 0; 
00068         context = 0;
00069         for(typename tCst::const_iterator it = cst.begin(), end=cst.end(); it != end; ++it){
00070                 if( it.visit() == 1 ){
00071                         typename tCst::size_type d = cst.depth(*it);
00072                         if( d >= k ){
00073                                 if( d == k )
00074                                         hk += cst.leaves_in_the_subtree(*it) * H0(*it, cst);    
00075                                 ++context;
00076                                 it.skip_subtree();
00077                         }
00078                 }
00079         }       
00080         hk /= cst.size();
00081         return hk;

00088 typename Csa::size_type get_char_pos(typename Csa::size_type idx, typename Csa::size_type d, const Csa &csa){
00089         if( d == 0 )
00090                 return idx;
00091         // if we have to apply \f$\LF\f$ or \f$\Phi\f$ more
00092         // than 2*d times to calc csa(csa[idx]+d), we opt to 
00093         // apply \f$ \Phi \f$ d times
00094         if( (csa.sa_sample_dens - 1) + (csa.isa_sample_dens - 1) > 2*d ){
00095                 for(typename Csa::size_type i=0; i < d; ++i)
00096                         idx = csa.psi[idx];
00097                 return idx;     
00098         }
00099         return csa( csa[idx] + d );
