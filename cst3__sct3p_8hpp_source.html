00267                 // Get the previous smaller value.
00268                 /* 
00269                  * \par Time complexity 
00270                  *    \f$ \Order{\frac{\sigma}{w}} \f$, where w=64 is the word size, can be implemented in \f$\Order{1}\f$ with rank and select
00271                  */  
00272                 inline size_type psv(size_type i, size_type ipos, size_type cipos, size_type &psvpos, size_type &psvcpos)const{
00273                         if( cipos >= m_bp.size() - m_sigma ){ // if lcp[i]==0 => psv is the 0th index by definition
00274                                 psvpos = 0;
00275                                 psvcpos = m_bp.size()-1;
00276                                 return 0;
00277                         }
00278                         size_type r0 = cipos - m_bp_support.rank(cipos); // index of clothing parenthesis in first_child bp
00279                         size_type next_first_child = 0;
00280                         const uint64_t *p = m_first_child.data() + (r0>>6);
00281                         uint64_t w = (*p) >> (r0&0x3F);
00282                         if( w ){ // if w!=0
00283                                 next_first_child = cipos + bit_magic::r1BP(w);
00284                                 if( cipos == next_first_child and m_bp[next_first_child+1] ){
00285                                         psvpos = m_bp_support.enclose(ipos);
00286                                         psvcpos = m_bp_support.find_close(psvpos);
00287                                         return m_bp_support.rank( psvpos )-1;
00288                                 }
00289                         }else{
00290                                 cipos += 64-(r0&0x3F); 
00291                                 ++p;
00292                                 while( !(w=*p) ){ // while w==0
00293                                         ++p;
00294                                         cipos += 64;
00295                                 }
00296                                 next_first_child = cipos + bit_magic::r1BP(w);
00297                         }
00298                         if( !m_bp[next_first_child+1] ){ // if next parenthesis is a closing one
00299                                 psvcpos = next_first_child+1;
00300                                 psvpos = m_bp_support.find_open(psvcpos);
00301                                 return m_bp_support.rank( psvpos )-1;
00302                         }
00303                         else{
00304                                 psvpos = m_bp_support.enclose( m_bp_support.find_open(next_first_child) );
00305                                 psvcpos = m_bp_support.find_close(psvpos);
00306                                 return m_bp_support.rank( psvpos )-1;
00307                         }
00308                 }

00700                 size_type degree(const node_type &v)const{
00701                         if( is_leaf(v) ) // if v is a leave, v has no child
00702                                 return 0;
00703                         // v is not a leave: v has at least two children
00704                         size_type r = get_pos_of_closing_parenthesis_of_the_first_l_index(v);
00705 /*                      if( m_bp[r-1] ){// if there exists no next l-index
00706                                 return 2;
00707                         }
00708 */                      
00709                         size_type r0 = r - m_bp_support.rank(r);
00710                         const uint64_t *p = m_first_child.data() + (r0>>6);
00711                         uint8_t offset = r0&0x3F;
00712 
00713                         uint64_t w = (*p) & bit_magic::Li1Mask[offset];
00714                         if( w ){
00715                                 return offset-bit_magic::l1BP(w)+1;
00716                         }else if( m_first_child.data() == p ){ // w==0 and we are in the first word
00717                                 return offset+2; // da bit_magic::l1BP(w)=-1 sein muesste
00718                         }else{
00719                                 size_type res = offset+2;
00720                                 while( p > m_first_child.data() ){
00721                                         w = *(--p);
00722                                         if( 0==w )
00723                                                 res += 64;
00724                                         else{
00725                                                 return res + (63-bit_magic::l1BP(w));
00726                                         }
00727                                 }
00728                                 return res + (w==0);
00729                         }
00730                 }
