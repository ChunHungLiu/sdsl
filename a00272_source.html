00523 {

00526         return;

00535     uint64_t n=0, vals=0;

00541             vec[n+1] = 0;
00542             vec[++n] = k >> (int_width-(offset-64));
00543             offset -= 64;

00548     for (typename int_vector_type::size_type i=0; i < n64;) {
00549         for (uint64_t ii=0; ii < n and i < n64; ++ii,++i) {
