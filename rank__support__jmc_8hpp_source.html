00113 inline void rank_support_jmc::init(const int_vector<1>* v)

00116     if (m_v == NULL) return;

00123     m_blockrank.set_int_width(12);
00124     m_blockrank.resize((m_v->capacity()>>6) + (0==(m_v->size()&0x3F)));     // n/64 + 2*loglog 64
00125     m_superblockrank.set_int_width(m_logn);
00126     m_superblockrank.resize((m_blockrank.size()+63)>>6);

00130     size_type cnt = 0, blockcnt = 0, wcnt = 0;
00131     const uint64_t* data = m_v->data();
00132     size_type i;
00133     for (i = 1; i < (m_v->capacity()>>6) ; ++i) {
00134         wcnt = bit_magic::b1Cnt(*data);
00135         ++data;
00136         blockcnt += wcnt;
00137         cnt      += wcnt;
00138         if ((i & 0x3F) == 0) {
00139             m_superblockrank[i>>6] = cnt;
00140             blockcnt = 0;
00141         }
00142         m_blockrank[i] = blockcnt;
00143     }
00144     if (0 == (m_v->size()&0x3F)) {
