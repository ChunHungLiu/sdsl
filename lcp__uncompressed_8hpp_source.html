00206 template<uint8_t width>
00207 template<class Text, class Sa>
00208 void lcp_uncompressed<width>::construct(const Text& text, const Sa& sa)
00209 {
00210     if (sa.size() == 0) {
00211         return;
00212     }
00213     m_lcp = int_vector<width>(sa.size(), 0, width);
00214 
00215     // use Kasai algorithm to compute the lcp values
00216     typename int_vector<width>::size_type sa_1 = sa(0), l=0;
00217     for (typename int_vector<width>::size_type i=0,j=0; i < sa.size(); ++i) {
00218         if (l) --l;
00219         if (sa_1) {
00220             j = sa[sa_1-1];
00221             while (i+l < sa.size() and j+l < sa.size() and text[i+l]==text[j+l]) ++l;
00222         } else {
00223             l = 0;
00224         }
00225         m_lcp[ sa_1 ] = l;
00226         sa_1 = sa.psi[sa_1];
00227     }
00228 }
