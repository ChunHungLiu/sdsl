00228 void lcp_kurtz<width>::construct(const Text& text, const Sa& sa)

00231         return;
00232     }
00233     m_small_lcp = int_vector<8>(sa.size());
00234     tVPII big_values;
00235 
00236     // use Kasai algorithm to compute the lcp values
00237     typename Sa::size_type sa_1 = sa(0), l=0, max_l=0, max_big_idx=0;
00238     for (typename Sa::size_type i=0,j=0; i < sa.size(); ++i) {
00239         if (l) --l;
00240         if (sa_1) {
00241             j = sa[sa_1-1];
00242             while (i+l < sa.size() and j+l < sa.size() and text[i+l]==text[j+l]) ++l;
00243         } else {
00244             l = 0;
00245         }
00246         if (l < 255) {
00247             m_small_lcp[ sa_1 ] = l;
00248         } else {
00249             m_small_lcp[ sa_1 ] = 255;
00250             big_values.push_back(tPII(sa_1, l));
00251             if (l > max_l) max_l = l;
00252             if (sa_1 > max_big_idx) max_big_idx = sa_1;
00253         }
00254         sa_1 = sa.psi[sa_1];
00255     }
00256 
00257     sort(big_values.begin(), big_values.end());
00258     m_big_lcp           = int_vector<width>(big_values.size(), 0, bit_magic::l1BP(max_l)+1);
00259     m_big_lcp_idx       = int_vector<width>(big_values.size(), 0, bit_magic::l1BP(max_big_idx)+1);
00260     for (tVPII::size_type i=0; i<big_values.size(); ++i) {
