00220                 return;

00222         m_small_lcp = int_vector<8>(sa.size());
00223         tVPII big_values;       
00224 
00225         // use Kasai algorithm to compute the lcp values
00226         typename Sa::size_type sa_1 = sa(0), l=0, max_l=0, max_big_idx=0;       
00227         for(typename Sa::size_type i=0,j=0; i < sa.size();++i){
00228                 if(l) --l;
00229                 if( sa_1 ){
00230                         j = sa[sa_1-1]; 
00231                         while(i+l < sa.size() and j+l < sa.size() and text[i+l]==text[j+l]) ++l;
00232                 }
00233                 else{
00234                         l = 0;
00235                 }
00236                 if( l < 255 ){
00237                         m_small_lcp[ sa_1 ] = l;
00238                 }else{
00239                         m_small_lcp[ sa_1 ] = 255;
00240                         big_values.push_back( tPII(sa_1, l) );
00241                         if( l > max_l ) max_l = l;
00242                         if( sa_1 > max_big_idx ) max_big_idx = sa_1;
00243                 }       
00244                 sa_1 = sa.psi[sa_1];
00245         }
